  public ModelAndView listAccounts(){
        var m = new ModelAndView();
        List<Account> accounts = accountService.getAccounts();
        m.addObject("accounts",accounts);
        Double SUM = 0.0;
        for(Account a : accounts){
            SUM = SUM + a.getBalance();
        }
        m.addObject("SUM",SUM);
        m.setViewName("account/list");
        return m;

    }
